#include <avr/io.h>
// 960 us N=64 C=16M
// 16M/64 = 25*10^4
// 1/(25*10^4) = 4u
//960u/4u = 240
//240-1 = 239
int main(void){
	DDRB |= 1<<5
	TCCR0A = 0;
	TCCR0B = 4; //timer1 CTC = 100
	OCR0A = 239;
    while (1) {
		while((TIFR0 & (1<<OCF0A))==0){}
			TIFR0 = (1<<OCF0A);
			PORTB ^= 1<<5; // toggle PB5
	}
	return 0;
}
