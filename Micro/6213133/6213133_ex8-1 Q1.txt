// example 8-11
// program to monitor PORTB.1 and turn PORTC.7 on if PORTB.1 is HIGH


#include <avr/io.h>    		

int main(void) 
{
	DDRB = DDRB & 0b11111100; // PB.0,1 is input			
	DDRD = 0xFF;
	while(1) // 00 01 10 11
	{
		if((PINB & 0b00000010) != 0)  //11,10
			if((PINB & 0b00000011) != 3) //10
				PORTD = '3';
			else //11
				PORTD = '4';
		else
		//if((PINB & 0b00000010) != 2)  //01,00
			if((PINB & 0b00000001) != 0) //01
				PORTD = '2';
			else //00
				PORTD = '1';
				
	} 

	return 0;
}
