#include <avr/io.h>

int main(void)
{
	DDRD |= (1<<6);	//OC0A as output
	OCR0A = 51; // 256*0.2 = 51.2
	TCCR0A = (1<<COM0A1)|(1<<WGM01)|(1<<WGM00); //Fast PWM, non-inverted
	// 16M/(7812*256) = 8
	TCCR0B = 0x02; //fast PWM = 010
	// 1 loop = 3*0.043 = 0.129 ms
	// 1/0.129m = 7800 -> correct
	while(1);
}
