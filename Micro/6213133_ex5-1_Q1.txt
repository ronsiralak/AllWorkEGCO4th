	
L1:	
	ldi r16,0x55

	OUT	PORTB,R16	;send data to port B
	CALL DELAY	
	OUT	PORTC,R16	
	CALL DELAY
	OUT	PORTD,R16	
	CALL DELAY

	ldi r16,0xAA

	OUT	PORTB,R16
	CALL DELAY
	OUT	PORTC,R16
	CALL DELAY
	OUT	PORTD,R16
	CALL DELAY

	RJMP	L1		;keep doing this

	;-------------------delay subroutine---------------
	.ORG 0x300	; put the subroutine far away at 0x300	
DELAY:
	LDI R20,0xFF	
AGAIN:
	NOP				
	NOP
	DEC R20			
	BRNE AGAIN		
	RET
